#: from:
#:
#: http://www.thonky.com/qr-code-tutorial/mask-patterns/

    lambda row,col: row % 2 == 0,
    lambda row,col: col % 3 == 0,
    lambda row,col: (row + col) % 3 == 0,
    lambda row,col: ((row // 2) + (col // 3)) % 2 == 0,
